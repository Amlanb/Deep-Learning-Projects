In[6]:
# Handle outliers in 'Age' using the IQR method
Q1 = df["Age"].quantile(0.25)
Q3 = df["Age"].quantile(0.75)
IQR = Q3 - Q1
lower_bound = Q1 - 1.5 * IQR
upper_bound = Q3 + 1.5 * IQR
df = df[(df["Age"] >= lower_bound) & (df["Age"] <= upper_bound)]
